  [1, 60, 10],
  [2, 100, 20],
  [3, 120, 30],

function knapsack(item, cap) {
  let memo = []

  for (let i = 0; i <= item.length; i++) {
    memo[i] = []
  }

  for (let i = 0; i < item.length + 1; i++) {
    for (let j = 0; j <= cap; j++) {
      if (i === 0 || j === 0) {
        // 1
        memo[i][j] = 0
      } else if (item[i - 1][2] > j) {
        // 2
        memo[i][j] = memo[i - 1][j]
      } else {
        // 3
        memo[i][j] = Math.max(
          memo[i - 1][j],
          memo[i - 1][j - item[i - 1][2]] + item[i - 1][1],
        )
      }
    }
  }
  return memo[item.length][cap]
}
